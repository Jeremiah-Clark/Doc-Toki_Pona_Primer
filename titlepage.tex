% titlepage.tex — Title page layout (pure LaTeX, edit per project)
%
% Available commands (populated from master.yaml via template.tex):
%   \doctitle       — Document title
%   \docauthor      — Author name
%   \docdate        — Date string
%   \docversion     — Version string (only defined if version: is set in YAML)
%   \doclogo        — Path to logo image (only defined if logo: is set in YAML)
%   \docdisclaimer  — Disclaimer text (only defined if disclaimer: is set in YAML)

\begin{titlepage}
  \thispagestyle{empty}

  % Version stamp — top-right corner
  \ifdefined\docversion
    \begin{flushright}
      \vspace*{-4em}
      {\small\sffamily\color{black!60} v\docversion}
    \end{flushright}
    \vspace{-3em}
  \fi

  \begin{center}
    \rule{\linewidth}{0.4pt}
    \vspace{1em}

    {\Huge\sffamily \doctitle}

    \vspace{1em}
    {\large\sffamily \docauthor}

    \vspace{0.5em}
    {\docdate}

    \vspace{1em}

    \ifdefined\doclogo
      \includegraphics[width=0.5\textwidth]{\doclogo}
    \fi

    \vfill

    \ifdefined\docdisclaimer
      \begin{important}
      \docdisclaimer
      \end{important}
    \fi

  \end{center}
\end{titlepage}
